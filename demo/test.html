Usage-

            
<span data-i18="weight"></span>

From javascript

            
var onLoad = function ( ){
    console.log( window.translate.translation_text( 'usage' ) );
    console.log( window.translate.translation_text( 'hello', {"username":"nohponex", "email" : "nohponex@gmail.com"} ) );
}
//Initialize
$(function() {
    window.translate = new $.fn.Translate( 
        { project_id : 1, language : 'gr', API_KEY : 'API_KEY', cache_duration:10000, onLoad : onLoad }
    );
});
